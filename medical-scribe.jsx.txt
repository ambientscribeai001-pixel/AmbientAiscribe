// ─── DEVICE DETECTION ─────────────────────────────────────────────────────────
function useDevice() {
  const [d, setD] = useState(() => window.innerWidth < 640 ? "mobile" : window.innerWidth < 1024 ? "tablet" : "desktop");
  useEffect(() => {
    const fn = () => setD(window.innerWidth < 640 ? "mobile" : window.innerWidth < 1024 ? "tablet" : "desktop");
    window.addEventListener("resize", fn);
    return () => window.removeEventListener("resize", fn);
  }, []);
  return d;
}

// ─── API CLIENT ───────────────────────────────────────────────────────────────
function useApi(token) {
  return useCallback(async (method, path, body = null) => {
    if (DEMO_MODE) throw new Error("Demo mode — backend not connected");
    const res = await fetch(`${BACKEND_URL}${path}`, {
      method,
      headers: { "Content-Type": "application/json", ...(token ? { Authorization: `Bearer ${token}` } : {}) },
      ...(body ? { body: JSON.stringify(body) } : {}),
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.message || `API ${res.status}`);
    return data;
  }, [token]);
}

// ─── CLAUDE API ───────────────────────────────────────────────────────────────
async function claudeJSON(system, user) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ model: "claude-sonnet-4-6", max_tokens: 1000, system, messages: [{ role: "user", content: user }] }),
  });
  const d = await res.json();
  if (!res.ok) throw new Error(d.error?.message || "Claude API error");
  const raw = d.content.map(b => b.text || "").join("");
  return JSON.parse(raw.replace(/```json\s*/gi, "").replace(/```\s*/gi, "").trim());
}

// ─── WAVEFORM ─────────────────────────────────────────────────────────────────
function Waveform({ active, playTime, total=60, height=44 }) {
  const ref=useRef(null), bars=useRef(Array.from({length:48},()=>Math.random()*0.55+0.15)), frame=useRef(0), raf=useRef(null);
  useEffect(()=>{
    const c=ref.current; if(!c) return;
    const ctx=c.getContext("2d");
    const draw=()=>{
      const W=c.parentElement?.offsetWidth||300; c.width=W; c.height=height;
      ctx.clearRect(0,0,W,height);
      const bw=W/bars.current.length, pf=total>0&&playTime!=null?playTime/total:-1;
      bars.current.forEach((h,i)=>{
        const a=active?Math.min(1,Math.max(0.05,h+Math.sin(frame.current*0.09+i*0.38)*0.28)):h*0.22;
        const bh=a*height*0.88;
        ctx.fillStyle=pf>=0&&i/bars.current.length<=pf?"#6366f1":active?`rgba(99,102,241,${0.18+a*0.55})`:"#1e2535";
        ctx.beginPath(); ctx.roundRect(i*bw+1,(height-bh)/2,Math.max(bw-2,1),bh,2); ctx.fill();
      });
      frame.current++; raf.current=requestAnimationFrame(draw);
    };
    raf.current=requestAnimationFrame(draw);
    return ()=>cancelAnimationFrame(raf.current);
  },[active,playTime,total,height]);
  return <canvas ref={ref} style={{width:"100%",height,borderRadius:8,background:"#0b0e16",border:"1px solid #1e2535",display:"block"}}/>;
}

  return <div style={{display:"flex",flexDirection:"column",alignItems:"center",gap:2}}>
    <svg width={size} height={size} style={{transform:"rotate(-90deg)"}}>
      <circle cx={size/2} cy={size/2} r={r} fill="none" stroke="#1e2535" strokeWidth={4}/>
      <circle cx={size/2} cy={size/2} r={r} fill="none" stroke={col} strokeWidth={4} strokeDasharray={`${d} ${c-d}`} strokeLinecap="round" style={{transition:"stroke-dasharray 1s ease"}}/>

}
